	\begin{frame}[fragile]{Justification}

		\only<1>{
			\begin{itemize}
				\item \textbf{Parallel and distributed nature} of lightweight manycores
				\item Memory restrictions do not support a complete virtual environment like VMs
				\item Explore lighter forms of virtualization as at the OS level
				\item Integrate virtualization into a distributed OS designed
					for lightweight manycores
			\end{itemize}
		}
